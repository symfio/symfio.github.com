Some plugins may register the worker which will be called during the unloading process.

 
  register: (worker) ->
    @workers.unshift worker

  unload: (callback) ->
    @once "unloaded", callback if typeof callback is "function"

    async.series @workers, =>
      @emit "unloaded"

createInstance = ->
  new Unloader
